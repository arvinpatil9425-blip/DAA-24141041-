import java.util.PriorityQueue;

public class OptimalMergePattern {

    public static int minMergeCost(int[] files) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();

        // Add all file sizes to the priority queue
        for (int size : files) {
            pq.add(size);
        }

        int totalCost = 0;

        // Merge two smallest files until one file remains
        while (pq.size() > 1) {
            int first = pq.poll();
            int second = pq.poll();
            int merged = first + second;
            totalCost += merged;
            pq.add(merged);
        }

        return totalCost;
    }

    public static void main(String[] args) {
        int[] fileSizes = {20, 30, 10, 5};  // Sizes of sorted files
        int cost = minMergeCost(fileSizes);
        System.out.println("Minimum total merge cost: " + cost);
    }
}
